005package org.anarres.tftp.server.netty;
006
007import io.netty.bootstrap.Bootstrap;
008import io.netty.channel.Channel;
009import io.netty.channel.ChannelFuture;
010import io.netty.channel.ChannelFutureListener;
011import io.netty.channel.ChannelHandlerContext;
012import io.netty.channel.ChannelInboundHandlerAdapter;
013import io.netty.util.ReferenceCountUtil;
014import java.net.InetSocketAddress;
015import javax.annotation.Nonnull;
016import org.anarres.tftp.protocol.engine.TftpTransfer;
017import org.anarres.tftp.protocol.packet.TftpErrorCode;
018import org.anarres.tftp.protocol.packet.TftpErrorPacket;
019import org.anarres.tftp.protocol.packet.TftpPacket;
020import org.anarres.tftp.protocol.packet.TftpRequestPacket;
021import org.anarres.tftp.protocol.resource.TftpData;
022import org.anarres.tftp.protocol.resource.TftpDataProvider;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026/**
027 *
028 * @author shevek
029 */
030public class TftpServerHandler extends ChannelInboundHandlerAdapter {
031
032    private static final Logger LOG = LoggerFactory.getLogger(TftpServerHandler.class);
033    private final TftpPipelineInitializer.SharedHandlers sharedHandlers;
034    private final TftpDataProvider provider;
035
036    public TftpServerHandler(@Nonnull TftpPipelineInitializer.SharedHandlers sharedHandlers, @Nonnull TftpDataProvider provider) {
037        this.sharedHandlers = sharedHandlers;
038        this.provider = provider;
039    }
040
041    @Override
042    public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
043        try {
044            final TftpPacket packet = (TftpPacket) msg;
045            Channel channel = ctx.channel();
046
047            switch (packet.getOpcode()) {
048                case RRQ: {
049                    TftpRequestPacket request = (TftpRequestPacket) packet;
050                    TftpData source = provider.open(request.getFilename());
051                    if (source == null) {
052                        ctx.writeAndFlush(new TftpErrorPacket(packet.getRemoteAddress(), TftpErrorCode.FILE_NOT_FOUND), ctx.voidPromise());
053                        // ctx.writeAndFlush(new TftpErrorPacket(packet.getRemoteAddress(), TftpErrorCode.FILE_NOT_FOUND)).addListener(ChannelFutureListener.FIRE_EXCEPTION_ON_FAILURE);
054                    } else {
055                        TftpTransfer<Channel> transfer = new TftpReadTransfer(packet.getRemoteAddress(), source, request.getBlockSize());
056                        Bootstrap bootstrap = new Bootstrap()
057                                .group(ctx.channel().eventLoop())
058                                .channel(channel.getClass())
059                                // .localAddress(new InetSocketAddress(0))
060                                // .remoteAddress(packet.getRemoteAddress())
061                                .handler(new TftpPipelineInitializer(sharedHandlers, new TftpTransferHandler(transfer)));
062                        bootstrap.connect(packet.getRemoteAddress());/*.addListener(new ChannelFutureListener() {
063                            @Override
064                            public void operationComplete(ChannelFuture future) throws Exception {
065                                LOG.info("Connected for " + packet);
066                            }
067                        });*/
068                    }
069                    break;
070                }
071                case WRQ: {
072                    // LOG.warn("Unexpected TFTP " + packet.getOpcode() + " packet: " + packet);
073                    ctx.writeAndFlush(new TftpErrorPacket(packet.getRemoteAddress(), TftpErrorCode.PERMISSION_DENIED), ctx.voidPromise());
074                    break;
075                }
076                case ACK: {
077                    break;
078                }
079                case DATA: {
080                    LOG.warn("Unexpected TFTP " + packet.getOpcode() + " packet: " + packet);
081                    ctx.writeAndFlush(new TftpErrorPacket(packet.getRemoteAddress(), TftpErrorCode.ILLEGAL_OPERATION), ctx.voidPromise());
082                    break;
083                }
084                case ERROR: {
085                    LOG.error("Received TFTP error packet: " + packet);
086                    break;
087                }
088            }
089
090        } catch (Exception e) {
091            ctx.fireExceptionCaught(e);
092            throw e;
093        } finally {
094            ReferenceCountUtil.release(msg);
095        }
096    }
097
098    @Override
099    public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
100        ctx.flush();
101    }
102
103    @Override
104    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
105        LOG.error("Error on channel: " + cause, cause);
106        // LOG.error("Reported here: " + cause, new Exception("Here"));
107    }
108}
