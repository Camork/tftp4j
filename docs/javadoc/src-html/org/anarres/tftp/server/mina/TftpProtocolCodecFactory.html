005package org.anarres.tftp.server.mina;
006
007import java.io.IOException;
008import javax.annotation.Nonnull;
009import org.anarres.tftp.protocol.codec.TftpPacketDecoder;
010import org.anarres.tftp.protocol.packet.TftpPacket;
011import org.apache.mina.core.buffer.IoBuffer;
012import org.apache.mina.core.session.IoSession;
013import org.apache.mina.filter.codec.ProtocolCodecFactory;
014import org.apache.mina.filter.codec.ProtocolDecoder;
015import org.apache.mina.filter.codec.ProtocolDecoderOutput;
016import org.apache.mina.filter.codec.ProtocolEncoder;
017import org.apache.mina.filter.codec.ProtocolEncoderOutput;
018
019/**
020 *
021 * @author shevek
022 */
023public class TftpProtocolCodecFactory implements ProtocolCodecFactory {
024
025    private static TftpProtocolCodecFactory INSTANCE = new TftpProtocolCodecFactory();
026
027    /**
028     * Returns the singleton instance of {@link TftpProtocolCodecFactory}.
029     *
030     * @return The singleton instance of {@link TftpProtocolCodecFactory}.
031     */
032    @Nonnull
033    public static TftpProtocolCodecFactory getInstance() {
034        return INSTANCE;
035    }
036
037    private static class TftpEncoder implements ProtocolEncoder {
038
039        public static final TftpEncoder INSTANCE = new TftpEncoder();
040
041        @Override
042        public void encode(IoSession session, Object message, ProtocolEncoderOutput out) throws IOException {
043            TftpPacket packet = (TftpPacket) message;
044            IoBuffer buf = IoBuffer.allocate(packet.getWireLength());
045            packet.toWire(buf.buf());
046            buf.flip();
047            out.write(buf);
048        }
049
050        @Override
051        public void dispose(IoSession session) throws Exception {
052        }
053    }
054
055    @Nonnull
056    @Override
057    public ProtocolEncoder getEncoder(@Nonnull IoSession session) {
058        return TftpEncoder.INSTANCE;
059    }
060
061    private static class TftpDecoder extends TftpPacketDecoder implements ProtocolDecoder {
062
063        public static final TftpDecoder INSTANCE = new TftpDecoder();
064
065        @Override
066        public void decode(IoSession session, IoBuffer in, ProtocolDecoderOutput out) throws Exception {
067            TftpPacket packet = super.decode(session.getRemoteAddress(), in.buf());
068            out.write(packet);
069        }
070
071        @Override
072        public void finishDecode(IoSession session, ProtocolDecoderOutput out) throws Exception {
073        }
074
075        @Override
076        public void dispose(IoSession session) throws Exception {
077        }
078    }
079
080    @Nonnull
081    @Override
082    public ProtocolDecoder getDecoder(@Nonnull IoSession session) {
083        return TftpDecoder.INSTANCE;
084    }
085}
