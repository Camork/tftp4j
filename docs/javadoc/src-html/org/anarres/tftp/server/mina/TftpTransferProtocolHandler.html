005package org.anarres.tftp.server.mina;
006
007import javax.annotation.Nonnull;
008import org.anarres.tftp.protocol.packet.TftpPacket;
009import org.anarres.tftp.protocol.engine.TftpTransfer;
010import org.apache.mina.core.future.ConnectFuture;
011import org.apache.mina.core.future.IoFutureListener;
012import org.apache.mina.core.service.IoHandlerAdapter;
013import org.apache.mina.core.service.IoService;
014import org.apache.mina.core.session.IdleStatus;
015import org.apache.mina.core.session.IoSession;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019/**
020 *
021 * @author shevek
022 */
023// TODO: Re-code this to use a window of (say) 8 packets.
024public class TftpTransferProtocolHandler extends IoHandlerAdapter implements IoFutureListener<ConnectFuture> {
025
026    private static final Logger LOG = LoggerFactory.getLogger(TftpTransferProtocolHandler.class);
027    private final IoService connector;
028    private final TftpTransfer<IoSession> transfer;
029
030    public TftpTransferProtocolHandler(@Nonnull IoService connector, @Nonnull TftpTransfer<IoSession> transfer) {
031        this.connector = connector;
032        this.transfer = transfer;
033    }
034
035    @Override
036    public void operationComplete(ConnectFuture future) {
037        if (!future.isConnected()) {
038            connector.dispose(false);
039            return;
040        }
041        final IoSession session = future.getSession();
042        try {
043            // This does "real" I/O, so can really throw an exception.
044            transfer.open(session);
045        } catch (Exception e) {
046            session.close(true);
047        }
048    }
049
050    @Override
051    public void messageReceived(IoSession session, Object message) throws Exception {
052        TftpPacket packet = (TftpPacket) message;
053        transfer.handle(session, packet);
054    }
055
056    @Override
057    public void sessionIdle(IoSession session, IdleStatus status) throws Exception {
058        transfer.timeout(session);
059    }
060
061    @Override
062    public void exceptionCaught(IoSession session, Throwable cause) throws Exception {
063        LOG.error("Exception caught in session " + session + ": " + cause, cause);
064        session.close(true);
065    }
066
067    @Override
068    public void sessionClosed(IoSession session) throws Exception {
069        super.sessionClosed(session);
070        connector.dispose(false);
071        transfer.close(session);
072    }
073}
