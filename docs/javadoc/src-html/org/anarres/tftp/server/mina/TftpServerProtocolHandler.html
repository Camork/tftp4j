005package org.anarres.tftp.server.mina;
006
007import java.net.SocketAddress;
008import javax.annotation.Nonnull;
009import org.anarres.tftp.protocol.engine.TftpTransfer;
010import org.anarres.tftp.protocol.packet.TftpErrorCode;
011import org.anarres.tftp.protocol.packet.TftpErrorPacket;
012import org.anarres.tftp.protocol.packet.TftpPacket;
013import org.anarres.tftp.protocol.packet.TftpRequestPacket;
014import org.anarres.tftp.protocol.resource.TftpData;
015import org.anarres.tftp.protocol.resource.TftpDataProvider;
016import org.apache.mina.core.future.ConnectFuture;
017import org.apache.mina.core.service.IoHandlerAdapter;
018import org.apache.mina.core.session.IoSession;
019import org.apache.mina.filter.codec.ProtocolCodecFilter;
020import org.apache.mina.filter.logging.LoggingFilter;
021import org.apache.mina.transport.socket.nio.NioDatagramConnector;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025/**
026 *
027 * @author shevek
028 */
029public class TftpServerProtocolHandler extends IoHandlerAdapter {
030
031    private static final Logger LOG = LoggerFactory.getLogger(TftpServerProtocolHandler.class);
032    private final TftpDataProvider provider;
033
034    public TftpServerProtocolHandler(@Nonnull TftpDataProvider provider) {
035        this.provider = provider;
036    }
037
038    @Override
039    public void messageReceived(IoSession session, Object message) throws Exception {
040        TftpPacket packet = (TftpPacket) message;
041
042        SocketAddress address = session.getRemoteAddress();
043        LOG.info("Address is " + address);
044
045        switch (packet.getOpcode()) {
046            case RRQ: {
047                TftpRequestPacket request = (TftpRequestPacket) packet;
048                TftpData source = provider.open(request.getFilename());
049                if (source == null) {
050                    session.write(new TftpErrorPacket(address, TftpErrorCode.FILE_NOT_FOUND), address);
051                    session.close(false);
052                } else {
053                    final NioDatagramConnector connector = new NioDatagramConnector();
054                    TftpTransfer transfer = new TftpReadTransfer(address, source, request.getBlockSize());
055                    TftpTransferProtocolHandler handler = new TftpTransferProtocolHandler(connector, transfer);
056                    connector.getFilterChain().addLast("codec", new ProtocolCodecFilter(new TftpProtocolCodecFactory()));
057                    connector.getFilterChain().addLast("logger-packet", new LoggingFilter("tftp-transfer-packet"));
058                    connector.setHandler(handler);
059                    ConnectFuture future = connector.connect(address);
060                    future.addListener(handler);
061                }
062                break;
063            }
064            case WRQ: {
065                session.write(new TftpErrorPacket(address, TftpErrorCode.PERMISSION_DENIED), address);
066                session.close(false);
067                break;
068            }
069            case ACK: {
070                break;
071            }
072            case DATA: {
073                LOG.warn("Unexpected TFTP " + packet.getOpcode() + " packet: " + packet);
074                session.write(new TftpErrorPacket(address, TftpErrorCode.ILLEGAL_OPERATION), address);
075                session.close(false);
076                break;
077            }
078            case ERROR: {
079                TftpErrorPacket error = (TftpErrorPacket) packet;
080                LOG.error("Received TFTP error packet: " + error);
081                session.close(true);
082                break;
083            }
084        }
085    }
086
087    @Override
088    public void exceptionCaught(IoSession session, Throwable cause) throws Exception {
089        LOG.error("Exception caught in session " + session + ": " + cause, cause);
090    }
091}
