005package org.anarres.tftp.server.mina;
006
007import java.io.IOException;
008import java.net.InetSocketAddress;
009import javax.annotation.Nonnegative;
010import javax.annotation.Nonnull;
011import org.anarres.tftp.protocol.engine.AbstractTftpServer;
012import org.anarres.tftp.protocol.resource.TftpDataProvider;
013import org.apache.mina.filter.codec.ProtocolCodecFilter;
014import org.apache.mina.filter.logging.LoggingFilter;
015import org.apache.mina.transport.socket.DatagramSessionConfig;
016import org.apache.mina.transport.socket.nio.NioDatagramAcceptor;
017
018/**
019 *
020 * @author shevek
021 */
022public class TftpServer extends AbstractTftpServer {
023
024    private NioDatagramAcceptor acceptor;
025
026    public TftpServer(@Nonnull TftpDataProvider dataProvider, @Nonnegative int port) {
027        super(dataProvider, port);
028    }
029
030    public TftpServer(@Nonnull TftpDataProvider dataProvider) {
031        super(dataProvider, DEFAULT_SERVER_PORT);
032    }
033
034    @Override
035    public void start() throws IOException {
036        acceptor = new NioDatagramAcceptor();
037        acceptor.setDefaultLocalAddress(new InetSocketAddress(getPort()));
038        acceptor.getFilterChain().addLast("logger-data", new LoggingFilter("tftp-server-data"));
039        acceptor.getFilterChain().addLast("codec", new ProtocolCodecFilter(new TftpProtocolCodecFactory()));
040        acceptor.getFilterChain().addLast("logger-packet", new LoggingFilter("tftp-server-packet"));
041        acceptor.setHandler(new TftpServerProtocolHandler(getDataProvider()));
042        DatagramSessionConfig dcfg = acceptor.getSessionConfig();
043        dcfg.setReuseAddress(true);
044        // dcfg.setIdleTime(IdleStatus.BOTH_IDLE, 5);
045        acceptor.bind();
046    }
047
048    @Override
049    public void stop() throws IOException {
050        acceptor.dispose();
051    }
052}
