005package org.anarres.tftp.protocol.packet;
006
007import com.google.common.base.Objects;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnull;
010
011/**
012 *
013 * @author shevek
014 */
015public class TftpDataPacket extends TftpPacket {
016
017    public static final int BLOCK_SIZE = 512;
018    /** Cheat for an unsigned 2-byte value. */
019    private char blockNumber;
020    private ByteBuffer data;
021
022    public TftpDataPacket() {
023    }
024
025    public TftpDataPacket(char blockNumber, @Nonnull ByteBuffer data) {
026        this.blockNumber = blockNumber;
027        this.data = data;
028    }
029
030    @Override
031    public int getWireLength() {
032        return data.remaining() + 256;
033    }
034
035    @Override
036    public TftpOpcode getOpcode() {
037        return TftpOpcode.DATA;
038    }
039
040    public void setBlockNumber(char blockNumber) {
041        this.blockNumber = blockNumber;
042    }
043
044    public char getBlockNumber() {
045        return blockNumber;
046    }
047
048    @Nonnull
049    public ByteBuffer getData() {
050        return data;
051    }
052
053    public void setData(ByteBuffer data) {
054        this.data = data;
055    }
056
057    @Override
058    public void toWire(ByteBuffer buffer) {
059        super.toWire(buffer);
060        buffer.putChar(getBlockNumber());
061        buffer.put(getData().slice());
062    }
063
064    @Override
065    public void fromWire(ByteBuffer buffer) {
066        setBlockNumber(buffer.getChar());
067        byte[] tmp = new byte[buffer.remaining()];
068        buffer.get(tmp);
069        setData(ByteBuffer.wrap(tmp));
070    }
071
072    @Override
073    protected Objects.ToStringHelper toStringHelper() {
074        return super.toStringHelper()
075                .add("blockNumber", (int) getBlockNumber())
076                .add("data", data);
077    }
078}
