005package org.anarres.tftp.protocol.packet;
006
007import com.google.common.base.Charsets;
008import com.google.common.base.Objects;
009import com.google.common.base.Preconditions;
010import java.net.SocketAddress;
011import java.nio.ByteBuffer;
012import java.nio.charset.Charset;
013import javax.annotation.Nonnegative;
014import javax.annotation.Nonnull;
015
016/**
017 *
018 * @author shevek
019 */
020public abstract class TftpPacket {
021
022    // Substitute for netascii in the RFC. ISO_8859_1 is 8-bit transparent.
023    public static final Charset CHARSET = Charsets.ISO_8859_1;
024    private SocketAddress remoteAddress;
025
026    @Nonnull
027    public SocketAddress getRemoteAddress() {
028        return remoteAddress;
029    }
030
031    public void setRemoteAddress(@Nonnull SocketAddress remoteAddress) {
032        this.remoteAddress = Preconditions.checkNotNull(remoteAddress, "Remote address was null.");
033    }
034
035    /** At most this long, possibly less. */
036    @Nonnegative
037    public int getWireLength() {
038        return 256;
039    }
040
041    @Nonnull
042    public abstract TftpOpcode getOpcode();
043
044    @Nonnull
045    protected Objects.ToStringHelper toStringHelper() {
046        return Objects.toStringHelper(this).add("opcode", getOpcode());
047    }
048
049    public void toWire(@Nonnull ByteBuffer buffer) {
050        buffer.putShort(getOpcode().getCode());
051    }
052
053    /** This is called after the opcode has already been read. */
054    public abstract void fromWire(@Nonnull ByteBuffer buffer);
055
056    protected static void putString(@Nonnull ByteBuffer buffer, @Nonnull String text) {
057        buffer.put(text.getBytes(CHARSET));
058        buffer.put((byte) 0);
059    }
060
061    @Nonnull
062    protected static String getString(@Nonnull ByteBuffer buffer) {
063        int start = buffer.position();
064        int finish;
065        int end = -1;
066        for (finish = start; finish < buffer.limit(); finish++) {
067            if (buffer.get(finish) == 0) {
068                end = finish;
069                finish++;
070                break;
071            }
072        }
073        // We didn't find a zero byte.
074        if (end == -1)
075            end = buffer.limit();
076
077        byte[] bytes = new byte[end - start];
078        buffer.get(bytes);
079        buffer.position(finish);
080        return new String(bytes, CHARSET);
081    }
082
083    @Override
084    public String toString() {
085        return toStringHelper().toString();
086    }
087}
