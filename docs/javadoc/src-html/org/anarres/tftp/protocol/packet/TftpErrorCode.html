005package org.anarres.tftp.protocol.packet;
006
007import javax.annotation.Nonnull;
008
009/**
010 *
011 * @author shevek
012 */
013public enum TftpErrorCode {
014
015    UNDEFINED(0, "Not defined, see error message (if any)."),
016    FILE_NOT_FOUND(1, "File not found."),
017    PERMISSION_DENIED(2, "Access violation."),
018    DISK_FULL(3, "Disk full or allocation exceeded."),
019    ILLEGAL_OPERATION(4, "Illegal TFTP operation."),
020    UNKNOWN_TRANSFER_ID(5, "Unknown transfer ID."),
021    FILE_ALREADY_EXISTS(6, "File already exists."),
022    NO_SUCH_USER(7, "No such user.");
023    private final short code;
024    private final String description;
025
026    /* pp */ TftpErrorCode(int code, String description) {
027        this.code = (short) code;
028        this.description = description;
029    }
030
031    public short getCode() {
032        return code;
033    }
034
035    @Nonnull
036    public String getDescription() {
037        return description;
038    }
039
040    @Override
041    public String toString() {
042        return name() + "(" + getCode() + ", " + getDescription() + ")";
043    }
044}
