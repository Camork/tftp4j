005package org.anarres.tftp.protocol.packet;
006
007import javax.annotation.Nonnull;
008
009/**
010 *
011 * @author shevek
012 */
013public enum TftpOpcode {
014
015    // TODO: Implement opcode 6, options-response.
016    RRQ(1, "Read request"),
017    WRQ(2, "Write request"),
018    DATA(3, "Data"),
019    ACK(4, "Acknowledgment"),
020    ERROR(5, "Error"),
021    // http://tools.ietf.org/html/rfc2347
022    ACK_WITH_OPTIONS(6, "Ack with options");
023    private final short code;
024    private final String description;
025
026    /* pp */
027    TftpOpcode(int code, String description) {
028        this.code = (short) code;
029        this.description = description;
030    }
031
032    public short getCode() {
033        return code;
034    }
035
036    @Nonnull
037    public String getDescription() {
038        return description;
039    }
040
041    @Override
042    public String toString() {
043        return name() + "(" + getCode() + ", " + getDescription() + ")";
044    }
045
046    @Nonnull
047    public static TftpOpcode forCode(short code) {
048        for (TftpOpcode opcode : TftpOpcode.values())
049            if (opcode.getCode() == code)
050                return opcode;
051        throw new IllegalArgumentException("No such TFTP opcode " + code);
052    }
053}
