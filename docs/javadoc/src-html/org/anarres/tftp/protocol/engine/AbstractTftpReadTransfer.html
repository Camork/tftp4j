005package org.anarres.tftp.protocol.engine;
006
007import com.google.common.annotations.VisibleForTesting;
008import com.google.common.math.IntMath;
009import com.google.common.primitives.Chars;
010import org.anarres.tftp.protocol.packet.TftpDataPacket;
011import java.io.IOException;
012import java.math.RoundingMode;
013import java.net.SocketAddress;
014import java.nio.ByteBuffer;
015import javax.annotation.Nonnegative;
016import javax.annotation.Nonnull;
017import javax.annotation.OverridingMethodsMustInvokeSuper;
018import javax.annotation.concurrent.GuardedBy;
019import org.anarres.tftp.protocol.packet.TftpAckPacket;
020import org.anarres.tftp.protocol.packet.TftpErrorCode;
021import org.anarres.tftp.protocol.packet.TftpErrorPacket;
022import org.anarres.tftp.protocol.packet.TftpPacket;
023import org.anarres.tftp.protocol.resource.TftpData;
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026
027/**
028 *
029 * @author shevek
030 */
031public abstract class AbstractTftpReadTransfer<TftpTransferContext> extends AbstractTftpTransfer<TftpTransferContext> {
032
033    private static final Logger LOG = LoggerFactory.getLogger(AbstractTftpReadTransfer.class);
034    public static final int MAX_WINDOW_SIZE = 16;
035    public static final int MAX_RETRIES = 3;
036    private final TftpData source;
037    private final int blockSize;
038    private final int blockCount;
039    // Runtime
040    /** The next block to send, indexed from 0. */
041    @GuardedBy("lock")
042    private int sendBlock = 0;
043    @GuardedBy("lock")
044    private int sendWindow = MAX_WINDOW_SIZE;
045    /** The last block acked, indexed from 0. */
046    // We start at -2, then -1 on open, causing us to send 0.
047    @GuardedBy("lock")
048    private int recvBlock = -2;
049    @GuardedBy("lock")
050    private int recvRetry = 0;
051    private final Object lock = new Object();
052
053    public AbstractTftpReadTransfer(@Nonnull SocketAddress remoteAddress, @Nonnull TftpData source, @Nonnegative int blockSize) throws IOException {
054        super(remoteAddress);
055        this.source = source;
056        this.blockSize = blockSize;
057        this.blockCount = IntMath.divide(source.getSize() + 1, blockSize, RoundingMode.CEILING);
058    }
059
060    @Nonnull
061    public abstract ByteBuffer allocate(@Nonnull TftpTransferContext context, @Nonnegative int length);
062
063    /**
064     * @param blockNumber indexed from 0
065     */
066    @Nonnull
067    @GuardedBy("lock")
068    private TftpDataPacket newPacket(@Nonnull TftpTransferContext context, int blockNumber) throws IOException {
069        ByteBuffer buf = allocate(context, blockSize);
070        // Note that if length is 0, we still construct a "final" zero-length packet.
071        source.read(buf, blockNumber * blockSize);
072        buf.flip();
073        return new TftpDataPacket(Chars.checkedCast(blockNumber + 1), buf);
074    }
075
076    /**
077     * @param ackBlock indexed from 0
078     */
079    @VisibleForTesting
080    /* pp */ void ack(@Nonnull TftpTransferContext context, @Nonnegative int ackBlock) throws Exception {
081        // if (LOG.isDebugEnabled()) LOG.debug("<- Ack protocol-block {} (index {})", (ackBlock + 1), ackBlock);
082
083        synchronized (lock) {
084            if (ackBlock < recvBlock) {
085                LOG.warn("{}: Out of order ack {} < {} previously received", new Object[]{
086                    this, ackBlock, recvBlock
087                });
088                // An ack got out of order?
089            } else if (ackBlock == recvBlock) {
090                // 1) The client is re-acking last known block (client timeout)
091                if (recvRetry++ >= MAX_RETRIES) {
092                    LOG.error("{}: Retries exceeded {} at packet {}", this, AbstractTftpReadTransfer.MAX_RETRIES, ackBlock + 1);
093                    close(context);
094                    return;
095                } else {
096                    LOG.warn("{}: Retry {} of packet {}", this, recvRetry, ackBlock + 1);
097                    sendBlock = ackBlock + 1;
098                    sendWindow = Math.max(1, sendWindow >> 1);
099                }
100            } else if (ackBlock == blockCount - 1) {
101                // We are done.
102                close(context);
103                return;
104            } else {
105                // The first "open" ack goes through here.
106                // Let's assume that clients don't ack forwards if they're missing a block.
107                recvBlock = ackBlock;
108                recvRetry = 0;
109            }
110
111            // Make sure we don't run off the end of the data.
112            int sendLimit = Math.min(recvBlock + sendWindow, blockCount);
113            while (sendBlock < sendLimit)
114                send(context, newPacket(context, sendBlock++));
115        }
116    }
117
118    @Override
119    public void open(@Nonnull TftpTransferContext context) throws Exception {
120        ack(context, -1);
121        flush(context);
122    }
123
124    @Override
125    public void handle(@Nonnull TftpTransferContext context, @Nonnull TftpPacket packet) throws Exception {
126        switch (packet.getOpcode()) {
127            case ACK:
128                TftpAckPacket ack = (TftpAckPacket) packet;
129                ack(context, ack.getBlockNumber() - 1);
130                break;
131            case RRQ:
132            case WRQ:
133            case DATA:
134                LOG.warn("{}: Unexpected TFTP " + packet.getOpcode() + " packet: " + packet, this);
135                send(context, new TftpErrorPacket(packet.getRemoteAddress(), TftpErrorCode.ILLEGAL_OPERATION));
136                close(context);
137                break;
138            case ERROR:
139                TftpErrorPacket error = (TftpErrorPacket) packet;
140                LOG.error("{}: Received TFTP error packet: {}", this, error);
141                close(context);
142                break;
143        }
144        flush(context);
145    }
146
147    @Override
148    public void timeout(@Nonnull TftpTransferContext context) throws Exception {
149        ack(context, recvBlock);
150    }
151
152    @Override
153    @OverridingMethodsMustInvokeSuper
154    public void close(@Nonnull TftpTransferContext context) throws Exception {
155        source.close();
156    }
157}
