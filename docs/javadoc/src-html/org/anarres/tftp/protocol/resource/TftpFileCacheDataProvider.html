005package org.anarres.tftp.protocol.resource;
006
007import com.google.common.cache.CacheBuilder;
008import com.google.common.cache.CacheLoader;
009import com.google.common.cache.LoadingCache;
010import com.google.common.cache.Weigher;
011import com.google.common.io.Files;
012import java.io.File;
013import java.io.IOException;
014import java.util.concurrent.TimeUnit;
015import javax.annotation.Nonnegative;
016import javax.annotation.Nonnull;
017
018/**
019 *
020 * @author shevek
021 */
022public class TftpFileCacheDataProvider extends TftpFileChannelDataProvider {
023
024    public static final long DEFAULT_SIZE = 64L * 1024 * 1024;
025    private final LoadingCache<File, TftpByteArrayData> cache;
026
027    /**
028     * @param cacheSize The cache size in bytes.
029     */
030    public TftpFileCacheDataProvider(@Nonnull String prefix, @Nonnegative long cacheSize) {
031        super(prefix);
032        this.cache = CacheBuilder.newBuilder()
033                .weigher(new Weigher<Object, TftpByteArrayData>() {
034            public int weigh(Object key, TftpByteArrayData value) {
035                return value.getSize();
036            }
037        }).maximumWeight(cacheSize)
038                .expireAfterAccess(10, TimeUnit.MINUTES)
039                .softValues()
040                .build(new CacheLoader<File, TftpByteArrayData>() {
041            @Override
042            public TftpByteArrayData load(File key) throws Exception {
043                byte[] data = Files.toByteArray(key);
044                return new TftpByteArrayData(data);
045            }
046        });
047    }
048
049    /**
050     * @param cacheSize The cache size in bytes.
051     */
052    public TftpFileCacheDataProvider(@Nonnegative long cacheSize) {
053        this(DEFAULT_PREFIX, cacheSize);
054    }
055
056    /** Uses a 64 Mb cache. */
057    public TftpFileCacheDataProvider(@Nonnull String prefix) {
058        this(prefix, DEFAULT_SIZE);
059    }
060
061    /** Uses a 64 Mb cache. */
062    public TftpFileCacheDataProvider() {
063        this(DEFAULT_PREFIX, DEFAULT_SIZE);
064    }
065
066    @Override
067    public TftpData open(String filename) throws IOException {
068        File file = toFile(filename);
069        if (file == null)
070            return null;
071        return cache.getUnchecked(file);
072    }
073}
