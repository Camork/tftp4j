005package org.anarres.tftp.protocol.resource;
006
007import com.google.common.base.Preconditions;
008import java.io.IOException;
009import java.nio.ByteBuffer;
010import javax.annotation.Nonnull;
011
012/**
013 *
014 * @author shevek
015 */
016public class TftpByteBufferData extends AbstractTftpData {
017
018    private final ByteBuffer data;
019
020    public TftpByteBufferData(@Nonnull ByteBuffer data) {
021        this.data = data;
022    }
023
024    @Override
025    public int getSize() {
026        return data.remaining();
027    }
028
029    @Override
030    public int read(ByteBuffer out, int offset) throws IOException {
031        Preconditions.checkPositionIndex(offset, getSize(), "Illegal data offset.");
032        int length = Math.min(getSize() - offset, out.remaining());
033        ByteBuffer slice = data.slice();    // We might not be reading from a full buffer.
034        slice.position(offset).limit(offset + length);
035        out.put(slice);
036        return length;
037    }
038}
