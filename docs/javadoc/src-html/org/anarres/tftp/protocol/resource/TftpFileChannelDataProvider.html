005package org.anarres.tftp.protocol.resource;
006
007import com.google.common.primitives.Ints;
008import java.io.File;
009import java.io.IOException;
010import java.io.RandomAccessFile;
011import java.nio.channels.FileChannel;
012import java.nio.file.StandardOpenOption;
013import javax.annotation.CheckForNull;
014import javax.annotation.Nonnull;
015import org.codehaus.mojo.animal_sniffer.IgnoreJRERequirement;
016
017/**
018 *
019 * @author shevek
020 */
021public class TftpFileChannelDataProvider extends AbstractTftpDataProvider {
022
023    public static final String DEFAULT_PREFIX = "/tftproot";
024    private final String prefix;
025
026    public TftpFileChannelDataProvider(@Nonnull String prefix) {
027        this.prefix = prefix;
028    }
029
030    public TftpFileChannelDataProvider() {
031        this(DEFAULT_PREFIX);
032    }
033
034    @Nonnull
035    public String getPrefix() {
036        return prefix;
037    }
038
039    /**
040     * Prepends the prefix to the filename, then returns an ordinary file, or null.
041     */
042    @CheckForNull
043    protected File toFile(@Nonnull String filename) {
044        String path = toPath(getPrefix(), filename);
045        if (path == null)
046            return null;
047        File file = new File(path);
048        if (!file.isFile())
049            return null;
050        if (!file.canRead())
051            return null;
052        return file;
053    }
054
055    @Override
056    // @IgnoreJRERequirement
057    public TftpData open(String filename) throws IOException {
058        File file = toFile(filename);
059        if (file == null)
060            return null;
061        RandomAccessFile raf = new RandomAccessFile(file, "r");
062        FileChannel channel = raf.getChannel();
063        // FileChannel channel = FileChannel.open(file.toPath(), StandardOpenOption.READ);
064        return new TftpFileChannelData(channel, Ints.checkedCast(channel.size()));
065    }
066}
